Overview

Modify echosrv-v2.c to use fork() to act as a concurrent echo server (ability to serve multiple clients at once). You may test your server using the telnet command as shown in the lecture, but eventually you should create echocli.c to act as a client.

   }

   struct sockaddr_in sa = {
      .sin_family = AF_INET,
      .sin_port = htons(PORT),
      .sin_addr = htonl(INADDR_ANY)
   };

   if(bind(sockfd, (struct sockaddr*) &sa, sizeof(sa)) < 0) {
      perror("Problem with bind()");
      exit(EXIT_FAILURE);
   }

   if(listen(sockfd, BACKLOG) < 0) {
      perror("Problem with listen()");
      exit(EXIT_FAILURE);
   }

   int port = ntohs(sa.sin_port);
   char ip[INET_ADDRSTRLEN];
   if(inet_ntop(AF_INET, (struct sockaddr*) &sa.sin_addr, ip, INET_ADDRSTRLEN) == NULL) {
      fprintf(stderr, "Couldn't convert server address\n");
   } else {
      printf("Listening on %s:%d\n", ip, port);
   }

   for(;;) {
      struct sockaddr_in* clia = malloc(sizeof(struct sockaddr_in));
      socklen_t addrlen = sizeof(struct sockaddr_in);

         continue;

}

Issues

You may end up with an error bind(): Address already in use due to an expected TCP behavior upon server crash. When a connection is terminates in a strange manner, TCP will set a timer (typically configured at 60 seconds) before fully closing the connection. There are two ways around this, you either wait or you kill the process.

$ ./server
Problem with bind(): Address already in use
$ ss | egrep 9001
tcp   CLOSE-WAIT 6      0                         127.0.0.1:9001         127.0.0.1:39284
$ ps
    PID TTY          TIME CMD
  20060 pts/0    00:00:00 bash
  20178 pts/0    00:00:00 server
  20182 pts/0    00:00:00 server
  20323 pts/0    00:00:00 ps
$ kill 20178
$ ps
    PID TTY          TIME CMD
  20060 pts/0    00:00:00 bash
  20359 pts/0    00:00:00 ps
